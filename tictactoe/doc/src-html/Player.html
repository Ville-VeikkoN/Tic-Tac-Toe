001/**
002* Player sets the 'x' to gameboard.
003*
004* <p>Checks that the input is valid and if it is, 
005* then sets it to gameboard.
006*
007* @author      Ville-Veikko Nieminen
008* @version     2017.1209
009* @since       1.6     
010*/
011class Player {
012    /** 
013    * Vertical coordinate for the 'x' or '0'.
014    */
015    protected static int placeX;
016    /** 
017    * Horizontal coordinate for the 'x' or '0'
018    */
019    protected static int placeY;
020    /**
021    * Checks if the user input for the 'x' is valid.
022    *
023    * <p> Checks that there is only integers in the input
024    * and that the integers is separated with ','. Then
025    * checks that coordinates can be found in the gameboard.
026    * Returns true or false if the input is valid or not.
027    *
028    * @param a User input for the 'x' in String.
029    * @param gameboard Char array which is the gameboard.
030    *
031    * @return true or false depending if the input is valid or not.
032    */
033    public boolean isUserInputValid(String a, char [][] gameboard) {
034                
035        a = a+",";
036        boolean firstCoord = true;
037        int indexNum = 0;
038        placeX = -1;
039        placeY = -1;
040
041        while(indexNum < a.length()) {
042            String s = "";
043            char c = a.charAt(indexNum);
044            while(c != ',') {
045                s += c;
046                indexNum++;
047                c = a.charAt(indexNum);
048            }
049            try {
050                if(firstCoord == true) {
051                    placeX = Integer.parseInt(s);
052                    firstCoord = false;
053                } else {
054                    placeY = Integer.parseInt(s);
055                }
056                indexNum++;
057                
058                if(placeX>=0 && placeX<TicTacToe.getHeigth() && placeY>=0 && placeY<TicTacToe.getWidth() && gameboard[placeX][placeY] == ' ') {
059                    return true;
060                } 
061            }catch (NumberFormatException ex) {
062                System.out.println("Not a number or you did not use ',' to separate coordinates.");
063                return false;
064            }
065        }
066        return false;
067    }
068
069    /** 
070    * Set the 'x' to the gameboard.
071    *
072    * @param gameboard Char array which is the gameboard.
073    */
074    public void setX(char [][] gameboard) {
075
076        gameboard[placeX][placeY] = 'x';
077    }
078}

080/**
081* AI sets the '0' to the gameboard.
082*
083* <p> AI extends the Player class to get the latest 'x' coordinates.
084*
085* @author      Ville-Veikko Nieminen
086* @version     2017.1209
087* @since       1.6   
088*/
089class AI extends Player {
090    /**
091    * Tells if the '0' is set or not.
092    */
093    private static boolean is0Set;
094
095    /** 
096    * Chooses the place for computer ('0').
097    *
098    * @param gameboard Char array which is the gameboard.
099    */
100    public void set0(char [][] gameboard) {
101        int aiSetsIn;
102        int h = TicTacToe.getHeigth();
103        int w = TicTacToe.getWidth();
104        int howManytoWin = TicTacToe.getHowManytoWin();
105        is0Set = false;
106        if(howManytoWin >= 4) {
107            aiSetsIn = 3;
108        } else {
109            aiSetsIn = 2;                
110        }
111
112        aiDiagonal(gameboard,h,w,howManytoWin,aiSetsIn);
113        aiNegDiagonal(gameboard,h,w,howManytoWin,aiSetsIn);
114        aiRow(gameboard,h,w,howManytoWin,aiSetsIn);
115        aiVertRow(gameboard,h,w,howManytoWin,aiSetsIn);
116        aiRandom(gameboard,h,w);
117    }                  
118    /** 
119    * Checks if AI is needed.
120    * 
121    * <p>If there is right amount on x's in diagonal rows, then
122    * try to block the user from winning.
123    *
124    * @param gameboard Char array which is the gameboard.
125    * @param h  Heigth of the gameboard.
126    * @param w  Width of the gameboard.
127    * @param howManytoWin   Amount of chars in line order to win.
128    * @param aiSetsIn   Tells when AI sets in.
129    */
130    public void aiDiagonal(char [][] gameboard, int h, int w, int howManytoWin, int aiSetsIn) {
131        int amountOfXs = 0;
132        int x = 1;
133        while(placeX-x >= 0 && placeY-x >= 0 && gameboard[placeX-x][placeY-x] == 'x' && is0Set == false) {
134            amountOfXs++;
135            if(placeX+1 < h && placeY+1 < w && gameboard[placeX+1][placeY+1] == ' ' && amountOfXs >= howManytoWin-aiSetsIn) {                 
136                gameboard[placeX+1][placeY+1] = '0';
137                is0Set = true;
138            } else if (placeX-(x+1) >=0 && placeY-(x+1) >=0 && gameboard[placeX-(x+1)][placeY-(x+1)] == ' ' && amountOfXs >= howManytoWin-(aiSetsIn)) {
139                gameboard[placeX-(x+1)][placeY-(x+1)] = '0';
140                is0Set = true;
141            }
142            x++;
143        }
144        x=1;
145        amountOfXs=0;
146
147        while(placeX+x < h && placeY+x < w && gameboard[placeX+x][placeY+x] == 'x' && is0Set == false) {
148            amountOfXs++;
149            if(placeX-1 >= 0 && placeY-1 >= 0 && gameboard[placeX-1][placeY-1] == ' ' && amountOfXs >= howManytoWin-aiSetsIn) {
150                gameboard[placeX-1][placeY-1] = '0';
151                is0Set = true;
152            } else if (placeX+(x+1) < h && placeY+(x+1) <w && gameboard[placeX+(x+1)][placeY+(x+1)] == ' ' && amountOfXs >= howManytoWin-(aiSetsIn)) {
153                gameboard[placeX+(x+1)][placeY+(x+1)] = '0';
154                is0Set = true;
155            }
156            x++;
157        }
158    }
159    /** 
160    * Checks if AI is needed.
161    *
162    * <p>If there is right amount on x's in negative diagonal rows, then
163    * try to block the user from winning.
164    *
165    * @param gameboard Char array which is the gameboard.

167    * @param w  Width of the gameboard.
168    * @param howManytoWin   Amount of chars in line order to win.
169    * @param aiSetsIn   Tells when AI sets in.
170    */
171    public void aiNegDiagonal(char [][] gameboard, int h, int w, int howManytoWin, int aiSetsIn) {
172        int x = 1;
173        int amountOfXs = 0;
174        while(placeX+x < h && placeY-x >=0 && gameboard[placeX+x][placeY-x] == 'x' && is0Set == false) {
175            amountOfXs++;
176            if(placeX-1 >= 0 && placeY+1 < w && gameboard[placeX-1][placeY+1] == ' ' && amountOfXs >= howManytoWin-aiSetsIn) {
177                gameboard[placeX-1][placeY+1] = '0';
178                is0Set = true;
179            }else if(placeX+(x+1) <h && placeY-(x+1) >=0 && gameboard[placeX+(x+1)][placeY-(x+1)] == ' ' && amountOfXs >= howManytoWin-(aiSetsIn)) {
180                gameboard[placeX+(x+1)][placeY-(x+1)] = '0';
181                is0Set = true;
182            }
183            x++;
184        }
185        x=1;
186        amountOfXs=0;
187
188        while(placeX-x >= 0 && placeY+x < w && gameboard[placeX-x][placeY+x] == 'x' && is0Set == false) {
189            amountOfXs++;
190            if(placeX+1 < h && placeY-1 >= 0 && gameboard[placeX+1][placeY-1] == ' ' && amountOfXs >= howManytoWin-aiSetsIn) {
191                gameboard[placeX+1][placeY-1] = '0';
192                is0Set = true;
193            } else if (placeX-(x+1) >=0 && placeY+(x+1) <w && gameboard[placeX-(x+1)][placeY+(x+1)] == ' ' && amountOfXs >= howManytoWin-(aiSetsIn)) {
194                gameboard[placeX-(x+1)][placeY+(x+1)] = '0';
195                is0Set = true;
196            }
197            x++;
198        }
199    }
200    /** 
201    * Checks if AI is needed.
202    *
203    * <p>If there is right amount on x's in vertical rows, then
204    * try to block the user from winning.
205    *
206    * @param gameboard Char array which is the gameboard.
207    * @param h  Heigth of the gameboard.
208    * @param w  Width of the gameboard.
209    * @param howManytoWin   Amount of chars in line order to win.
210    * @param aiSetsIn   Tells when AI sets in.
211    */
212    public void aiVertRow(char [][] gameboard, int h, int w, int howManytoWin, int aiSetsIn) {
213        int x = 1;
214        int amountOfXs = 0;
215        while(placeX >= 0 && placeY-x >=0 && gameboard[placeX][placeY-x] == 'x' && is0Set == false) {
216            amountOfXs++;
217            if(placeX < h && placeY+1 < w && gameboard[placeX][placeY+1] == ' ' && amountOfXs >= howManytoWin-aiSetsIn) {
218                gameboard[placeX][placeY+1] = '0';
219                is0Set = true;
220            } else if (placeX >=0 && placeY-(x+1) >=0 && gameboard[placeX][placeY-(x+1)] == ' ' && amountOfXs >= howManytoWin-(aiSetsIn)) {
221                gameboard[placeX][placeY-(x+1)] = '0';
222                is0Set = true;
223            }
224            x++;
225        }
226         x=1;
227        amountOfXs = 0; 
228
229        while(placeX >= 0 && placeY+x < TicTacToe.getWidth() && gameboard[placeX][placeY+x] == 'x' && is0Set == false) {
230            amountOfXs++;
231            if(placeX < h && placeY-1 >= 0 && gameboard[placeX][placeY-1] == ' ' && amountOfXs >= howManytoWin-aiSetsIn) { 
232                gameboard[placeX][placeY-1] = '0';
233                is0Set = true;
234            } else if (placeX >=0 && placeY+(x+1) < w && gameboard[placeX][placeY+(x+1)] == ' ' && amountOfXs >= howManytoWin-(aiSetsIn)) {
235                gameboard[placeX][placeY+(x+1)] = '0';
236                is0Set = true;

239        }

241    /** 
242    * Checks if AI is needed.

246    *
247    * @param gameboard Char array which is the gameboard.
248    * @param h  Heigth of the gameboard.
249    * @param w  Width of the gameboard.

253    public void aiRow(char [][] gameboard, int h, int w, int howManytoWin, int aiSetsIn) {
254        int amountOfXs = 0;
255        int x = 1;
256        while(placeX-x >= 0 && placeY >=0 && gameboard[placeX-x][placeY] == 'x' && is0Set == false) {
257            amountOfXs++;
258            if(placeX+1 < h && placeY < w && gameboard[placeX+1][placeY] == ' ' && amountOfXs >= howManytoWin-aiSetsIn) {
259                gameboard[placeX+1][placeY] = '0';
260                is0Set = true;
261            } else if (placeX-(x+1) >=0 && placeY >=0 && gameboard[placeX-(x+1)][placeY] == ' ' && amountOfXs >= howManytoWin-(aiSetsIn)) {
262                gameboard[placeX-(x+1)][placeY] = '0';
263                is0Set = true;
264            } 
265            x++;   
266        }
267        x=1;
268        amountOfXs = 0;
269
270        while(placeX+x < h && placeY >=0 && gameboard[placeX+x][placeY] == 'x' && is0Set == false) {
271            amountOfXs++;
272            if(placeX-1 >=0 && placeY < w && gameboard[placeX-1][placeY] == ' ' && amountOfXs >= howManytoWin-aiSetsIn) {
273                gameboard[placeX-1][placeY] = '0';
274                is0Set = true;
275            } else if (placeX+(x+1) <h && placeY >=0 && gameboard[placeX+(x+1)][placeY] == ' ' && amountOfXs >= howManytoWin-(aiSetsIn)) {
276                gameboard[placeX+(x+1)][placeY] = '0';
277                is0Set = true;
278            } 
279            x++;  
280
281        }
282    }
283    /** 
284    * Puts '0' into the gameboard by random if needed.
285    *
286    * <p> If AI didn't choose the place for '0', then 
287    * aiRandom() chooses it randomly.
288    *
289    * @param gameboard Char array which is the gameboard.
290    * @param h  Heigth of the gameboard.
291    * @param w  Width of the gameboard.
292    */
293    public void aiRandom(char [][] gameboard, int h, int w) {
294        while(is0Set == false) {
295            placeX = (int)(Math.random()*h);
296            placeY = (int)(Math.random()*w);
297            if(gameboard[placeX][placeY] == ' ') {
298                is0Set = true;
299                gameboard[placeX][placeY] = '0';
300            }
301        }
302    }
